git status --сосотояние файлов в репозитории
git add . --добавление изменений всего репозитория в индек гита (точка указывает на то что вся папка индексируется)
git clone address repos -- клонирование репозитория по адресу
git log --список коммитов с хешом
git checkout хэш коммита --переход между коммитами по хешу из лога вышеы
git push origin main --зафиксировать в гитхабе
git pull origin main --забрать изменения тз гитхаба
git branch --список веток в локальном репозитории
git branch имя ветки--создание новой ветки
git checkout название ветки --переход между ветками
